Code example

          
  const getSumOfMarks = (diary) => Object.values(diary)
          .reduce((acc, mark) => acc + mark, 0);
  const son = ['first', 'second', 'third'];

  function whoseBicycle(diary1, diary2, diary3) {
    const marks = [diary1, diary2, diary3]
            .map((diary) => getSumOfMarks(diary));
    const maxMark = Math.max(...marks);
    const pos = marks.lastIndexOf(maxMark);

    return `I need to buy a bicycle for my ${son[pos]} son.`;
  }
